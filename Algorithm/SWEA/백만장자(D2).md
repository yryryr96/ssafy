```python
# 틀린풀이 (시간초과)
T = int(input())
for tc in range(1,T+1):
    N = int(input())
    stack = list(map(int,input().split()))
    temp = 0
    MAX = 0
    money = 0
    if stack[0] == max(stack):
        temp = 1
    else:

        while stack:
            A = max(stack)
            money += A*len(stack[:stack.index(A)])-sum(stack[:stack.index(A)])
            k = len(stack[:stack.index(A)+1])
            for i in range(k):
                stack.pop(0)

    if temp == 1:
        print(f'#{tc} 0')
    else:
        print(f'#{tc} {money}')
```

```python
# 뒤에서부터 스택으로 구현했는데 틀린거 -> 왜 틀린지 모르겠음
T = int(input())
for tc in range(1,T+1):
    N = int(input())
    stack = list(map(int,input().split()))
    money = 0
    ans = []

    while stack:
        k = stack.pop()
        if not stack:
            ans.append(k)
        else:
            if stack[-1] > k:
                ans.append(k)
                money += ans.pop(0) * len(ans) - sum(ans)
                ans.clear()
            else:
                ans.append(k)

    money += ans.pop(0) * len(ans) - sum(ans)
    print(f'#{tc} {money}')
```

```python
# 스택안쓰고 정답
T = int(input())
for tc in range(1,T+1):
    N = int(input())
    stack = list(map(int,input().split()))
    money = 0
    MAX = stack[-1]

    for i in range(N-1,-1,-1):
        if MAX < stack[i] :
            MAX = stack[i]
        else :
            money += MAX - stack[i]

    print(f'#{tc} {money}')
```
