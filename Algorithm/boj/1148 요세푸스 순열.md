# boj.1148 : 요세푸스 순열

```python
# 큐 개념 사용
import sys
input = sys.stdin.readline
N,K = map(int,input().split())
lst = [i for i in range(1,1+N)]
ans = []

i = K - 1
while lst:
    if i >= len(lst):
        while i >= len(lst):
            i = i - len(lst)
    ans.append(lst.pop(i))
    i = i + (K - 1)

lst1 = list(map(str,ans))
print(f'<{", ".join(lst1)}>')
```
